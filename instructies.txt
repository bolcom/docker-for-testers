- Bouw de nieuwe docker image: 
    docker build --rm -t testnetapp .
- Start de postgresql container:
    docker run --name pg -e POSTGRES_PASSWORD=testnet -d postgres
- Maak een tabel aan en zet er een waarde in:
      docker run -it --rm --link pg:pg postgres psql -h pg -U postgres
    password=testnet
    CREATE DATABASE testnet;
    \c testnet
    CREATE TABLE kv (key varchar(100) PRIMARY KEY, value varchar(100));
    INSERT INTO kv VALUES ('provider','testnet on pg');
    SELECT * FROM kv; -- Check that the data is really there
    \q  om af te sluiten
- Draai de testnetapp container zodat ie naar PG kijkt
    docker run -ti --link pg:pg -p 5000:5000 testnetapp
- Ga naar localhost:5000, check dat de waarde ook echt uit de DB komt
- Update de waarde in de DB
    docker run -it --rm --link pg:pg postgres psql -h pg -U postgres
    \c testnet
    UPDATE kv SET value='testnet on pg updated!' WHERE key='provider';
- Ga naar localhost:5000, check dat de waarde ook echt geupdate is
